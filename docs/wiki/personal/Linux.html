chmod

user-group-others
read = 4, write = 2, execute = 1
ex: chmod 600 keys

curl

curl https://some.url --output file

disk usage

du -sh . (-s is summary of disk usage, -h is human readable)
du -sh * summarizes each item in that folder
du -sh * | sort -rh | head -5 shows top 5 largest paths
(-r reverses, and -h does human sort, taking into account suffixes)

df -h shows free disk space on various devices

environment variables

export NAME=VALUE
echo $NAME

find folder

find . -type d -name 'foldernam'

mounting disk

lsblk lists block devices
sudo file -s /dev/xvdf shows file system for attached volumes
sudo mount /dev/xvdf /data mounts block

screen

Ctrl+a, d to detatch
screen -ls or Ctrl+a, "

shutting down

sudo shutdown now

SSH tunneling

ssh -L 5901:127.0.0.1:5901 -C -N -l ubuntu 35.153.231.137
-L specifies the port bindings (local/remote)
-C enables compression
-N says don't execute a remote command
-l specifies remote login name

Ubuntu updates

sudo apt-get update updates package lists
sudo apt-get upgrade upgrades every installed package

unzip

unzip -l zipfile shows contents
unzip zipfile -d folder unzips to folder
unzip -n zipfile doesn't overwrite
unzip -q zipfile hides output
